integer rows
integer coef
integer space
integer i
integer j

rows = 8

for i = 0; i < rows; i = i + 1

   for space = 1; space <= rows-i; space = space + 1
      Put "  " to output

   for j = 0; j <= i; j = j + 1

      if j==0 or i==0
         coef = 1
      else
         coef = coef*(i-j+1)/j

      Put coef to output
      Put "   " to output

   Put "\n" to output
